\vspace{-6mm}
\begin{eqnarray*}
x^{(0)} &=& x \\[1mm]
x^{(1)} &=& \sigma\!\left(W_1 x^{(0)} + b_1\right) \\[1mm]
x^{(2)} &=& \sigma\!\left(W_2 x^{(1)} + b_2\right) \\[1mm]
\hat{y} &=& W_3 x^{(2)} + b_3
\end{eqnarray*}

\vspace{1mm}
\hspace{1cm}$\sigma$: ReLU, tanh, sigmoid
