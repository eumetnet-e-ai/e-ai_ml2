\textbf{Graph definition}

\begin{itemize}
  \item Graph $G = (V,E)$
  \item Nodes $V$: entities
  \item Edges $E$: relations
\end{itemize}

\vspace{1mm}
\textbf{Node data}

\begin{itemize}
  \item Each node has features $x_i$
  \item Labels $y_i$ for supervision
  \item Features may be physical states
\end{itemize}

\end{column}

\begin{itemize}

  \item Structure lives in edges
