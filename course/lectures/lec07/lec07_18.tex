stream = ollama.chat( model="llama3", 
   messages=[...], stream=True )

accumulated = ""
handle = display(
  Markdown(""), display_id=True)

for chunk in stream:
   if "content" in chunk["message"]:
      accumulated += chunk["message"]\
      ["content"] 
      handle.update(Markdown(accumulated))
