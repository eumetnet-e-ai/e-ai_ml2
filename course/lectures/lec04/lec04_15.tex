\begin{codeonly}{Creating dataset and loader}
x = np.linspace(0, 2*np.pi, 1000)
y = np.sin(x)

x_t = torch.tensor(x).float().unsqueeze(1)
y_t = torch.tensor(y).float().unsqueeze(1)

data = TensorDataset(x_t, y_t)
loader = DataLoader(data,
                    batch_size=32,
                    shuffle=True)
\end{codeonly}
